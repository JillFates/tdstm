016    package org.jsecurity.grails;
017    
018    
019    import org.jsecurity.authz.Permission;
020    
021    import java.util.*;
022    
023    /**
024     * A basic permission class that can be used with the default DB realm
025     * provided with the plugin.
026     */
027    public class JsecBasicPermission implements Permission {
028        public static final String WILDCARD = "*";
029    
030        private String target;
031        private Set actions;
032        private String actionsString;
033    
034        /**
035         * Creates a new permission with the given actions. <b>Note</b> an
036         * action may not have any whitespace in it, or ',' or ';' characters.
037         * @param target The permission target. The meaning of 'target' is
038         * application dependent, for example it may be a file path, a URL,
039         * a controller name, or a component name.
040         * @param actions A string of actions separated by ',', ';', or
041         * whitespace. For example, 'view,edit', 'create, delete',
042         * 'create;    modify', 'view edit' are all valid two-action strings.
043         */
044        public JsecBasicPermission(String target, String actions) {
045            this.target = target;
046    
047            // Convert the delimited string of actions into a set.
048            this.actions = actionsStringToSet(actions);
049        }
050    
051        /**
052         * Creates a new permission with the given actions.
053         * @param target The permission target. The meaning of 'target' is
054         * application dependent, for example it may be a file path, a URL,
055         * a controller name, or a component name.
056         * @param actions A collection of action strings, for example 'view'
057         * 'edit', 'create', and 'delete'.
058         */
059        public JsecBasicPermission(String target, Collection actions) {
060            this.target = target;
061    
062            // Copy the provided collection of actions.
063            this.actions = new HashSet(actions);
064        }
065    
066        /**
067         * Returns an unmodifiable set of this permission's actions.
068         */
069        public Set getActions() {
070            return Collections.unmodifiableSet(this.actions);
071        }
072    
073        /**
074         * Returns this permission's actions as a comma-separated string,
075         * for example 'view,create,modify,delete'.
076         */
077        public String getActionsString() {
078            // Delayed initialisation of the actions string.
079            if (this.actionsString == null) {
080                // Convert the set of actions to a string so that a standard
081                // delimiter is always used.
082                this.actionsString = actionsToString(this.actions);
083            }
084    
085            return this.actionsString;
086        }
087    
088        /**
089         * Returns <code>true</code> if this permission implies the given
090         * one, otherwise <code>false</code>. This method checks that this
091         * permission has the same target as the given permission and a
092         * super-set of the actions. The wildcard matches all targets and
093         * all actions.
094         */
095        public boolean implies(Permission p) {
096            boolean implies = (p instanceof JsecBasicPermission);
097    
098            if (implies) {
099                // Check the target strings are the same.
100                JsecBasicPermission perm = (JsecBasicPermission) p;
101                if (this.target == null) {
102                    implies = perm.target == null;
103                }
104                else if (!this.target.equals(WILDCARD)) {
105                    implies = this.target.equals(perm.target);
106                }
107    
108                // Now check the actions (unless the current permission
109                // includes the wildcard, which means that is implies
110                // all other actions).
111                if (implies && !this.actions.contains(WILDCARD)) {
112                    implies = this.actions.containsAll(perm.actions);
113                }
114            }
115    
116            return implies;
117        }
118    
119        // Custom equals() method.
120        public boolean equals(Object obj) {
121            if (!(obj instanceof JsecBasicPermission)) {
122                return false;
123            }
124    
125            JsecBasicPermission p = (JsecBasicPermission) obj;
126            return (this.target == null ? p.target == null : this.target.equals(p.target)) &&
127                    this.actions.equals(p.actions);
128        }
129    
130        // Custom hashCode() method.
131        public int hashCode() {
132            int result;
133            result = (this.target != null ? this.target.hashCode() : 0);
134            result = 31 * result + this.actions.hashCode();
135            return result;
136        }
137    
138        /**
139         * Converts a delimiter-separated string of actions into a set of
140         * actions. Supported delimiters are ',', ';', and whitespace.
141         */
142        protected Set actionsStringToSet(String actions) {
143            return new HashSet(Arrays.asList(actions.split("[,;\\s][\\s]*")));
144        }
145    
146        /**
147         * Converts a collection of actions into a comma-separated string.
148         * For example, 'view,edit,create,delete'. Note that there is no
149         * whitespace in the resulting string.
150         */
151        protected String actionsToString(Collection actions) {
152            StringBuffer buffer = new StringBuffer(actions.size() * 10);
153            for (Iterator iter = actions.iterator(); iter.hasNext();) {
154                buffer.append(iter.next()).append(',');
155            }
156    
157            return buffer.substring(0, buffer.length() - 1);
158        }
159    }
